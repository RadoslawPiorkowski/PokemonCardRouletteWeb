make database: "pokemon"

!!! For better working application you have to change postgres max connections!!!

Find "postgresql.conf" and change in it "max_connections" to around 700
You can also change "shared_buffers" to 5000MB - 7000MB to make it working faster with large amount of cards	
Later you have to restart your server, for WINDOWS:
1-Open Run Window by Winkey + R
2-Type services.msc
3-Search Postgres service based on version installed.
4-Click stop, start or restart the service option.

For macOS:
1-On the top left of the menu bar you have the Postgres Icon
2-Click on it this opens a drop down menu
3-Click on Stop -> than click on start

Execute in query tool:
Create table GRACZ(
	ID_GRACZ SERIAL,
	NICK varchar (25),
	MAIL varchar(40),
	HASLO varchar(20),
	FREE DATE NOT NULL DEFAULT '2020-08-08',
	PRIMARY KEY(ID_Gracz)
);


CREATE TABLE KARTA (
	ID_KARTA int,
	NAZWA varchar(20),
	RZADKOSC varchar(20),
	GRAFIKA varchar(250),
	PRIMARY KEY(ID_KARTA)
);

CREATE TABLE GRACZ_KARTA(
	ID_GRACZ_KARTA serial,
	GRACZ_ID int,
	KARTA_ID int,
	PRIMARY KEY(ID_GRACZ_KARTA),
	FOREIGN KEY (GRACZ_ID) REFERENCES GRACZ (ID_GRACZ) ON DELETE SET NULL,
	FOREIGN KEY (KARTA_ID) REFERENCES KARTA (ID_KARTA) ON DELETE SET NULL
);

Create table WIADOMOSC(
	IMIE varchar (30),
	MAIL varchar(40),
	NUMER varchar(20),
	TEMAT varchar(60),
	TEKST varchar(2000)
);
